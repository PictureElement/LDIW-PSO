/*
 * Inputs:
 *  s, swarm size
 *  dim, dimensionality
 *  c1, c2, learning factors
 *  lo, up, search-space boundaries
 *
 * Output:
 *  g, global optimal solution
 */
 PSO (s, dim, c1, c2, lo, up) {

    // Initialization process
    for i ← 0 to s - 1 {
        xi ← rand (lo, up)
        pi ← xi
        if i = 1
            g ← pi
        if f (pi) < f (g)
            g ← pi
        vi ← rand (-|up - lo|, |up - lo|)
    }

    // Optimization process
    do
        for i ← 0 to s - 1 {
            for d ← 0 to dim - 1 {
                vi,d ← vi,d + c1 * rand (0, 1) * (pi,d - xi,d) + c2 * rand (0, 1) * (gd - xi,d)
                xi,d ← bound (xi,d + vi,d, lo, up)
            }
            if f (xi) < f (pi)
                pi ← xi
            if f (pi) < f (g)
                g ← pi
    while <criteria> are met

    return g
}

// Velocity bound
bound (x, lo, up) {
    if x < lo
        return lo
    else if x > up
        return up
    else
        return x
}
